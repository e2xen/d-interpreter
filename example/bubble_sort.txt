var bubble := func (arr, n) is
  for i in 1 .. n-1 loop
    var temp
    for j in 1 .. n-i loop
      if arr[j] > arr[j+1] then
        temp := arr[j]
        arr[j] := arr[j+1]
        arr[j+1] := temp
      end
      for i in 1 .. n loop
        print arr[i]
      end
      print " "
    end
  end
end
var arr := [221, 113, 37, 15, 11]
var n := 5
bubble (arr, n)
