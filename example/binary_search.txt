 var arr := [1, 4, 7, 15, 30, 102, 113]
 var first := 1
 var last := 7
 var key := readInt
 var mid := (first + last)/2
 var binary := func (arr, first, last, mid, key) is
   for i in first .. last loop
     if arr[mid] < key then
       first := mid + 1
     else
       if arr[mid] = key then
         return mid
       else
         last := mid - 1
       end
     end
     mid := (first + last)/2
     if first > last then
       return "Element is not found!"
     end
   end
 end
 print binary(arr, first, last, mid, key)
